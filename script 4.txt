
CREATE TABLE CLIENTE2 (ID_CL INT NOT NULL, NOMBRE_CL VARCHAR(30) 
NOT NULL,APATERNO_CL VARCHAR(30) NOT NULL, AMATERNO_CL VARCHAR(30), 
CALLE_CL VARCHAR(100) NOT NULL, COLONIA_CL VARCHAR(50));

CREATE TABLE PLATILLO2 (ID_PL INT NOT NULL, NOMBRE_PL VARCHAR(50) NOT NULL,
DESCRIPCION VARCHAR(100) NOT NULL, PRECIO INT NOT NULL, CANTIDAD INT NOT NULL);

CREATE TABLE POBLACION2 (ID_P INT NOT NULL, NOMBRE_P VARCHAR(30) NOT NULL, 
NUM_HABITANTES INT NOT NULL);

CREATE TABLE CENTRO2 (ID_C INT NOT NULL, NOMBRE_C VARCHAR(30), CALLE_C VARCHAR(30) NOT NULL,
COLONIA_C VARCHAR(30) NOT NULL, TELEFONO_C INT NOT NULL);

CREATE TABLE EMPLEADO2 (ID_E INT NOT NULL, RFC VARCHAR(20) NOT NULL, NOMBRE_E VARCHAR(30)
NOT NULL, APATERNO_E VARCHAR(30) NOT NULL, AMATERNO_E VARCHAR(30), CALLE_E VARCHAR(30)
NOT NULL, COLONIA_E VARCHAR(30) NOT NULL, TELEFONO INT NOT NULL);

CREATE TABLE FACTURA2 (ID_F INT NOT NULL, DIA_F INT NOT NULL, MES_F INT NOT NULL,
ANO_F INT NOT NULL);
